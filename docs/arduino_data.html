 35    def save_durations(self, side_one, side_two, type_durations):
 36        """
 37        Function to save durations into long term storage valve_durations.toml file.
 38        You must provide a type_durations paramter to specify where you are saving the durations.
 39        Can be either "selected" to set new primary durations, or "archive" to archive previous
 40        "selected" durations.
 41        """
 42        dur_path = system_config.get_valve_durations()
 43
 44        with open(dur_path, "r") as f:
 45            toml_file = toml.load(f)
 46
 47        if type_durations == "selected":
 48            selected_durations = toml_file["selected_durations"]
 49
 50            # toml only knows how to deal with python native types (list, int, etc)
 51            # so we convert np arrays to lists
 52            selected_durations["date_used"] = datetime.datetime.now()
 53            selected_durations["side_one_durations"] = side_one.tolist()
 54            selected_durations["side_two_durations"] = side_two.tolist()
 55
 56            # save the file with the updated content
 57            with open(dur_path, "w") as f:
 58                toml.dump(toml_file, f)
 59        elif type_durations == "archive":
 60            # find first archive not filled
 61            first_avail = self.find_first_not_filled(toml_file)
 62            match first_avail:
 63                case 1:
 64                    # 1 is available just insert
 65                    archive = toml_file["archive_1"]
 66
 67                    archive["filled"] = True
 68                    archive["date_used"] = datetime.datetime.now()
 69                    archive["side_one_durations"] = side_one.tolist()
 70                    archive["side_two_durations"] = side_two.tolist()
 71
 72                    with open(dur_path, "w") as f:
 73                        toml.dump(toml_file, f)
 74                case 2:
 75                    # move 1-> 2, insert 1
 76                    toml_file["archive_2"] = copy.deepcopy(toml_file["archive_1"])
 77
 78                    archive_1 = toml_file["archive_1"]
 79
 80                    archive_1["filled"] = True
 81                    archive_1["date_used"] = datetime.datetime.now()
 82                    archive_1["side_one_durations"] = side_one.tolist()
 83                    archive_1["side_two_durations"] = side_two.tolist()
 84
 85                    with open(dur_path, "w") as f:
 86                        toml.dump(toml_file, f)
 87
 88                case 3 | 0:
 89                    # case 3) 2 -> 3, 1-> 2, insert 1
 90                    # case 0 (no available archives) ) del / overwrite 3 (oldest), 2 -> 3, 1-> 2, insert 1
 91                    toml_file["archive_3"] = copy.deepcopy(toml_file["archive_2"])
 92                    toml_file["archive_2"] = copy.deepcopy(toml_file["archive_1"])
 93
 94                    archive_1 = toml_file["archive_1"]
 95
 96                    archive_1["filled"] = True
 97                    archive_1["date_used"] = datetime.datetime.now()
 98                    archive_1["side_one_durations"] = side_one.tolist()
 99                    archive_1["side_two_durations"] = side_two.tolist()
100
101                    with open(dur_path, "w") as f:
102                        toml.dump(toml_file, f)

104    def load_durations(
105        self, type_durations="selected_durations", reset_durations=False
106    ):
107        """
108        This method loads data in from the valve_durations.toml file located in assets. By default,
109        it will load the last used durations becuase this is what is most often utilized. Optionally,
110        the user can reset all these values to default values of 24125ms if the reset_durations
111        parameter flag is set to true.
112        """
113        valve_durations_toml = system_config.get_valve_durations()
114
115        with open(valve_durations_toml, "r") as f:
116            toml_file = toml.load(f)
117
118        file_durations = toml_file[type_durations]
119
120        # create 2 np arrays 8 n long with np.int32s
121        dur_side_one = np.full(VALVES_PER_SIDE, np.int32(0))
122        dur_side_two = np.full(VALVES_PER_SIDE, np.int32(0))
123
124        # keys here are names of durations i.e side_one_dur or side_two_dur, values
125        # are lists of durations, VALVES_PER_SIDE long. Will produce two lists
126        # VALVES_PER_SIDE long so that each valve in the rig is assigned a duration.
127        for key, value in file_durations.items():
128            if key == "side_one_durations":
129                for i, duration in enumerate(value):
130                    dur_side_one[i] = duration
131            elif key == "side_two_durations":
132                for i, duration in enumerate(value):
133                    dur_side_two[i] = duration
134
135        date_used = file_durations["date_used"]
136
137        return dur_side_one, dur_side_two, date_used

139    def load_schedule_indices(self):
140        stimuli_data = self.exp_data.stimuli_data
141        schedule_df = self.exp_data.program_schedule_df
142
143        num_stimuli = self.exp_data.exp_var_entries["Num Stimuli"]
144        num_trials = self.exp_data.exp_var_entries["Num Trials"]
145
146        sched_side_one = np.full(num_trials, np.int8(0))
147        sched_side_two = np.full(num_trials, np.int8(0))
148
149        # get unique stim names. this will give us all names on side one.
150        # finding the index of a given stimuli appearing in a trial on side
151        # one will easily allow us to find the paired index. We've already done
152        # this before in exp_process_data
153        names = list(stimuli_data.stimuli_vars.values())
154        unique_names = names[: num_stimuli // 2]
155
156        side_one_trial_stimuli = schedule_df["Port 1"].to_numpy()
157
158        for i, trial_stim in enumerate(side_one_trial_stimuli):
159            index = unique_names.index(trial_stim)
160            sd_two_index = self.exp_data.get_paired_index(index, num_stimuli)
161            sched_side_one[i] = index
162            sched_side_two[i] = sd_two_index
163
164        return sched_side_one, sched_side_two

175    def handle_licks(self, split_data, event_data, state, trigger):
176        side = None
177        valve_duration = None
178        match state:
179            case "TTC":
180                # split_data will have the below form for ttc
181                # 0(side)|67(duration)|6541(stamp_rel_to_start)|6541(stamp_rel_to_trial)
182                # 1|16|0|496570|41 from arduino
183                side = np.int8(split_data[0])
184                self.increment_licks(side, event_data)
185
186                lick_duration = np.int32(split_data[1])
187                time_rel_to_start = np.int32(split_data[2]) / 1000
188                time_rel_to_trial = np.int32(split_data[3]) / 1000
189
190                # insert the values held in licks for respective sides in a shorter variable name
191                side_one = self.exp_data.event_data.side_one_licks
192                side_two = self.exp_data.event_data.side_two_licks
193                # if 3 or more licks in a ttc time, jump straight to sample
194                if side_one > 2 or side_two > 2:
195                    trigger("SAMPLE")
196                self.record_event(
197                    side, lick_duration, time_rel_to_start, time_rel_to_trial, state
198                )
199
200            case "SAMPLE":
201                try:
202                    # 0|87|26064|8327|8327
203                    side = np.int8(split_data[0])
204                    self.increment_licks(side, event_data)
205
206                    lick_duration = np.int32(split_data[1])
207                    valve_duration = np.int32(split_data[2])
208                    time_rel_to_start = np.int32(split_data[3]) / 1000
209                    time_rel_to_trial = np.int32(split_data[4]) / 1000
210
211                    self.record_event(
212                        side,
213                        lick_duration,
214                        time_rel_to_start,
215                        time_rel_to_trial,
216                        state,
217                        valve_duration,
218                    )
219
220                except Exception as e:
221                    logging.error(f"IMPROPER DATA.... IGNORING.....{e}")

252    def process_data(self, source, data, state, trigger):
253        """
254        Process data received from the Arduino.
255        """
256        event_data = self.exp_data.event_data
257        split_data = data.split("|")
258        try:
259            if split_data[0] == "MOTOR":
260                # data will arrive in the following format
261                # MOTOR|DOWN|2338|7340|7340
262                # MOTOR|MOVEMENT|DURATION|END_TIME_REL_TO_PROG_START|END_TIME_REL_TO_PROG_TRIAL_START
263                event_data = self.exp_data.event_data
264
265                current_trial = self.exp_data.current_trial_number
266
267                event_data.insert_row_into_df(
268                    current_trial,
269                    None,
270                    np.int32(split_data[2]),
271                    (np.int32(split_data[3]) / 1000),
272                    (np.int32(split_data[4]) / 1000),
273                    f"MOTOR {split_data[1]}",
274                )
275            elif split_data[0] == "0" or split_data[0] == "1":
276                self.handle_licks(split_data, event_data, state, trigger)
277
278        except Exception as e:
279            logging.error(f"Error processing data from {source}: {e}")
280            raise
