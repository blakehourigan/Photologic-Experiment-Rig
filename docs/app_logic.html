app_logic

'app_logic' is the primary module in the Photologic-Experiment-Rig codebase. It contains the StateMachine class which holds the logic for each 'state' an experiment can be in.

Before performing any actions, the StateMachine class initializes the 'TkinterApp' class, which launches a tkinter root and allows for a GUI to be created for the program.

This module is launched from main to make restarting the program easier, which is done by destroying the instance of state machine and launcing a new one.

 37class StateMachine(TkinterApp):
 38    """
 39    This class is the heart of the program. It inherits TkinterApp to create a gui, instantiate model (data) classes, and the arduino controller,
 40    then defines and handles program state transitions. It coordinates co-operation between view (gui) and models (data). State transitions are
 41    handled by the 'trigger' function.
 42    """
 43
 44    def __init__(self, result_container):
 45        # init the 'super' parent class which is the gui
 46        super().__init__()
 47        # default state is idle
 48        self.state = "IDLE"
 49        self.prev_state = None
 50        # used to make decision of whether to restart the program or just terminate
 51        self.app_result = result_container
 52        # define state transitions that the program can possibly take
 53        self.transitions = {
 54            ("IDLE", "GENERATE SCHEDULE"): "GENERATE SCHEDULE",
 55            (
 56                "GENERATE SCHEDULE",
 57                "IDLE",
 58            ): "IDLE",
 59            ("IDLE", "START"): "START PROGRAM",
 60            ("IDLE", "RESET"): "RESET PROGRAM",
 61            ("STOP PROGRAM", "RESET"): "RESET PROGRAM",
 62            ("START PROGRAM", "ITI"): "ITI",
 63            ("ITI", "DOOR OPEN"): "OPENING DOOR",
 64            ("OPENING DOOR", "TTC"): "TTC",
 65            ("TTC", "SAMPLE"): "SAMPLE",  # -> if rat engages in trial
 66            ("TTC", "TRIAL END"): "TRIAL END",  # -> if rat does NOT engage
 67            ("SAMPLE", "TRIAL END"): "TRIAL END",
 68            ("TRIAL END", "ITI"): "ITI",  # -> save trial data, update gui, door up
 69            ("ITI", "STOP"): "STOP PROGRAM",  # can move to 'stop' state in all states
 70            ("OPENING DOOR", "STOP"): "STOP PROGRAM",
 71            ("TTC", "STOP"): "STOP PROGRAM",
 72            ("SAMPLE", "STOP"): "STOP PROGRAM",
 73            ("TRIAL END", "STOP"): "STOP PROGRAM",
 74        }
 75
 76        # don't start the mainloop until AFTER the gui is setup so that the app_result property
 77        # is available if reset is desired
 78        self.main_gui.mainloop()

 80    def trigger(self, event):
 81        """This function takes a requested state and decides if the transition from this state is allowed."""
 82        transition = (self.state, event)
 83        self.prev_state = self.state
 84        new_state = None
 85
 86        logger.info(f"state transition -> {transition}")
 87
 88        # checking if the attemped transition is valid according to the table
 89        if transition in self.transitions:
 90            new_state = self.transitions[transition]
 91        else:
 92            # if the key is not in the transition table, handle rejection based on desired state
 93            self.reject_actions(event)
 94
 95        # if the key was in the transition table, and the new state wants to reset the program, that means this is a valid action at this time. MAKE SURE the
 96        # user REALLY wants to do that
 97        if new_state == "RESET PROGRAM":
 98            response = GUIUtils.askyesno(
 99                "============WARNING============",
100                "THIS ACTION WILL ERASE ALL DATA CURRENTLY STORED FOR THIS EXPERIMENT... ARE YOU SURE YOU WANT TO CONTINUE?",
101            )
102            # if true, go ahead with the reset, if false return to prev_state
103            if not response:
104                new_state = self.prev_state
105
106        # if attempt to start program with no experiment schedule, tell user to do that
107        if new_state == "START PROGRAM":
108            if self.exp_data.program_schedule_df.empty:
109                GUIUtils.display_error(
110                    "Experiement Schedule Not Yet Generated!!!",
111                    "Please generate the program schedule before attempting to start the experiment. You can do this by clicking the 'Valve / Stimuli' button in the main screen.",
112                )
113                new_state = self.prev_state
114
115        # if we have a new state, perform the associated action for that state
116        if new_state:
117            logger.info(f"new state -> {new_state}")
118            self.execute_state(new_state)

120    def execute_state(self, new_state: str):
121        """
122        takes the key for the transition table (from_state, to_state) and resulting_state, and executes
123        the action corresponding to new_state, based upon key. if the action is defined under state_target,
124        that action will be taken in a new thread to avoid overloading the main tkinter thread. otherwise
125        it is executed immediately in the main thread.
126        """
127        thread_target = None
128
129        match new_state:
130            case "START PROGRAM":
131
132                def thread_target():
133                    StartProgram(
134                        self.exp_data,
135                        self.main_gui,
136                        self.arduino_controller,
137                        self.trigger,
138                    )
139            case "RESET PROGRAM":
140                ResetProgram(self.main_gui, self.app_result, self.arduino_controller)
141            case "STOP PROGRAM":
142                StopProgram(self.main_gui, self.arduino_controller, self.trigger)
143            case "GENERATE SCHEDULE":
144                # because this will run in main thread, we need to return early to avoid self.state
145                # assignment confusion
146                self.state = new_state
147                GenerateSchedule(
148                    self.main_gui,
149                    self.arduino_controller,
150                    self.process_queue,
151                    self.trigger,
152                )
153                return
154            case "ITI":
155
156                def thread_target():
157                    InitialTimeInterval(
158                        self.exp_data,
159                        self.main_gui,
160                        self.arduino_controller,
161                        self.prev_state,
162                        new_state,
163                        self.trigger,
164                    )
165            case "OPENING DOOR":
166
167                def thread_target():
168                    OpeningDoor(
169                        self.exp_data,
170                        self.main_gui,
171                        self.arduino_controller,
172                        new_state,
173                        self.trigger,
174                    )
175            case "TTC":
176
177                def thread_target():
178                    TimeToContact(
179                        self.exp_data,
180                        self.arduino_controller,
181                        self.main_gui,
182                        new_state,
183                        self.trigger,
184                    )
185            case "SAMPLE":
186
187                def thread_target():
188                    SampleTime(
189                        self.exp_data,
190                        self.main_gui,
191                        self.arduino_controller,
192                        new_state,
193                        self.trigger,
194                    )
195            case "TRIAL END":
196
197                def thread_target():
198                    TrialEnd(
199                        self.exp_data,
200                        self.main_gui,
201                        self.arduino_controller,
202                        self.prev_state,
203                        self.trigger,
204                    )
205
206        self.prev_state = self.state
207        self.state = new_state
208
209        if thread_target:
210            threading.Thread(target=thread_target).start()

212    def process_queue(self, data_queue):
213        """
214        Process the data queue. Pulls in data from the thread that is reading data from the arduinos constantly. if there is anything
215        in the queue at time of function call, we will stay here until its all been dealt with.
216        """
217        arduino_data = self.exp_data.arduino_data
218        try:
219            while not data_queue.empty():
220                source, data = data_queue.get()
221                arduino_data.process_data(source, data, self.state, self.trigger)
222
223            # run this command every 250ms
224            queue_ps_id = self.main_gui.after(
225                250, lambda: self.process_queue(data_queue)
226            )
227            self.main_gui.scheduled_tasks["PROCESS QUEUE"] = queue_ps_id
228        except Exception as e:
229            logging.error(f"Error processing data queue: {e}")
230            raise

232    @staticmethod
233    def reject_actions(event):
234        match event:
235            case "RESET":
236                GUIUtils.display_error(
237                    "CANNOT PERFORM THIS ACTION",
238                    "RESET cannot be performed during runtime. Stop the program to reset.",
239                )
240            # if key is not in table but we try to start (for example, STOP PROGRAM -> START) we do NOT want to allow this until a full reset has been completed
241            case "START":
242                GUIUtils.display_error(
243                    "CANNOT PERFORM THIS ACTION",
244                    "Experiement has already been executed. Before running another, you need to RESET the application to ensure all data is reset to defaults",
245                )
246            case _:
247                return

333                5000, lambda: self.finalize_program()
334            )

336        except Exception as e:
337            logging.error(f"Error stopping program: {e}")
338            raise

341        """
342        Finalize the program by telling arduino to send the door up/down timestamps that it has, reset both Arduino boards,

344        """
345        try:
346            # stop the arduino listener so that the program can shut down
