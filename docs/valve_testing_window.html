451    def verify_variables(self, original_data) -> bool:
452        # wait until all data is on the wire ready to be read (4 bytes),
453        # then read it all in quick succession
454        while self.arduino_controller.arduino.in_waiting < 4:
455            pass
456

474    def verify_schedule(self, len_sched, sent_schedule) -> bool:
475        # np array used for verification later
476        received_sched = np.zeros((len_sched,), dtype=np.int8)
477
478        while self.arduino_controller.arduino.in_waiting < len_sched:
479            pass
480
481        for i in range(len_sched):
482            received_sched[i] = int.from_bytes(
483                self.arduino_controller.arduino.read(), byteorder="little", signed=False
484            )
485
486        if np.array_equal(received_sched, sent_schedule):
487            logger.info(f"===TESTING SCHEDULE VERIFIED as ==> {received_sched}===")
488            logger.info("====================BEGIN TESTING NOW====================")
489            return True
490        else:
491            msg = "====INCORRECT SCHEDULE DETECTED, TRY AGAIN===="
492            GUIUtils.display_error("TRANSMISSION ERROR", msg)
493            logger.error(msg)
494            return False

496    def send_schedules(self):
497        """
498        Calculate schedule lengths so arduino knows how many entried to look for on each schedule
499        receive operation. valve_selections is a np_array equivalent to the full test schedule
500        lineup. valves in schedule for side on come first, then side two.
501        """
502        len_sched_one = np.count_nonzero(self.valve_selections[:VALVES_PER_SIDE])
503        len_sched_two = np.count_nonzero(self.valve_selections[VALVES_PER_SIDE:])
504
505        len_sched_one = np.int8(len_sched_one)
506        len_sched_two = np.int8(len_sched_two)
507        len_sched = len_sched_one + len_sched_two
508
509        # because self.actuations is a IntVar, we use .get() method to get its value
510        valve_acuations = np.int16(self.actuations.get())
511
512        # schedule is TOTAL_VALVES long to accomodate testing of up to all valves at one time.
513        # if we are not testing them all, filter out the zero entries (valves we are not testing)
514        schedule = self.valve_selections[self.valve_selections != 0]
515
516        # decrement each element in the array by one to move to zero-indexing. We moved to 1
517        # indexing becuase using zero indexing does not work with the method of toggling valves
518        # defined above
519        schedule -= 1
520
521        self.side_one_tests = schedule[:len_sched_one]
522        self.side_two_tests = schedule[len_sched_one:]
523
524        len_sched_one = len_sched_one
525        len_sched_two = len_sched_two
526        valve_acuations = valve_acuations
527
528        self.arduino_controller.send_valve_durations()
529        if self.window_mode == WindowMode.TESTING:
530            ###====SENDING TEST COMMAND====###
531            command = "TEST VOL\n".encode("utf-8")
532            self.arduino_controller.send_command(command)
533
534        elif self.window_mode == WindowMode.PRIMING:
535            self.prime_running = True
536
537            command = "PRIME VALVES\n".encode("utf-8")
538            self.arduino_controller.send_command(command)
539
540        ###====SENDING TEST VARIABLES (len side one, two, num_actuations per test)====###
541        data_bytes = (
542            len_sched_one.tobytes()
543            + len_sched_two.tobytes()
544            + valve_acuations.tobytes()
545        )
546        self.arduino_controller.send_command(data_bytes)
547
548        # verify the data
549        if not self.verify_variables(data_bytes):
550            return
551
552        ###====SENDING SCHEDULE====###
553        sched_data_bytes = schedule.tobytes()
554        self.arduino_controller.send_command(sched_data_bytes)
555
556        if not self.verify_schedule(len_sched, schedule):
557            return
558
559        schedule += 1
560
561        valves = ",".join(str(valve) for valve in schedule)
562        if self.window_mode == WindowMode.TESTING:
563            ####### ARDUINO HALTS HERE UNTIL THE GO-AHEAD IS GIVEN #######
564            # inc each sched element by one so that it is 1-indexes and matches the valve labels
565            test_confirmed = GUIUtils.askyesno(
566                "CONFIRM THE TEST SCHEDULE",
567                f"Valves {valves}, will be tested. Review the test table to confirm schedule and timings. Each valve will be actuated {valve_acuations} times. Ok to begin?",
568            )
569
570            ### include section to manually modify timings
571            if test_confirmed:
572                self.bind("<<event0>>", self.testing_complete)
573                self.bind("<<event1>>", self.take_input)
574                threading.Thread(target=self.run_test).start()
575            else:
576                self.abort_test()
577        elif self.window_mode == WindowMode.PRIMING:
578            priming_confirmed = GUIUtils.askyesno(
579                "CONFIRM THE ACTION",
580                f"Valves {valves}, will be primed (opened and closed) {valve_acuations} times. Ok to begin?",
581            )
582            if priming_confirmed:
583                # in case of prime, we send 0 to mean continue or start
584                # and we send a 1 at any point to abort or cancel the prime
585                start = np.int8(0).tobytes()
586                self.arduino_controller.send_command(command=start)
587
588                if isinstance(self.valve_test_button, tk.Button):
589                    self.valve_test_button.configure(
590                        text="STOP Priming",
591                        bg="gold",
592                        command=lambda: self.stop_priming(),
593                    )
