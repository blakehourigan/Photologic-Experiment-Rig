103    def generate_schedule(self) -> bool:
104        """
105        This is a primary function of the program. This function generates the pseudo random schedule for what stimuli will be presented in which
106        trials. It generates the intervals for ITI, TTC, and sample time using base time plus generated random +/- variations as given by user
107        in interface.
108        """
109        try:
110            # set number of trials based on number of stimuli, and number of trial blocks set by user
111            num_stimuli = self.exp_var_entries["Num Stimuli"]
112
113            if num_stimuli > 8 or num_stimuli < 2 or num_stimuli % 2 != 0:
114                GUIUtils.display_error(
115                    "NUMBER OF STIMULI EXCEEDS CURRENT MAXIMUM",
116                    "Program is currently configured for a minumim of 2 and maximum of 8 TOTAL vavles. You must have an even number of valves. If more are desired, program configuration must be modified.",
117                )
118                return False
119
120            num_trial_blocks = self.exp_var_entries["Num Trial Blocks"]
121
122            self.exp_var_entries["Num Trials"] = (num_stimuli // 2) * num_trial_blocks
123
124            self.create_random_intervals()
125
126            pairs = self.create_trial_blocks()
127
128            # stimuli_1 & stimuli_2 are lists that hold the stimuli to be introduced for each trial on their respective side
129            stimuli_side_one, stimuli_side_two = self.generate_pairs(pairs)
130
131            # args needed -> stimuli_1, stimuli_2
132            # these are lists of stimuli for each trial, for each side respectivelyc:w
133            self.build_frame(
134                stimuli_side_one,
135                stimuli_side_two,
136            )
137            return True
138
139        except Exception as e:
140            logger.error(f"Error generating program schedule {e}.")
141            return False

143    def create_random_intervals(self) -> None:
144        try:
145            num_trials = self.exp_var_entries["Num Trials"]
146
147            final_intervals = [None, None, None]
148
149            interval_keys = ["ITI_var", "TTC_var", "sample_var"]
150
151            random_interval_keys = [
152                "ITI_random_entry",
153                "TTC_random_entry",
154                "sample_random_entry",
155            ]
156
157            # for each type of interval (iti, ttc, sample), generate final interval times
158            for i in range(len(interval_keys)):
159                base_val = self.interval_vars[interval_keys[i]]
160                # this makes an array that is n=num_trials length of base_val
161                base_intervals = np.repeat(base_val, num_trials)
162
163                # get plus minus value for this interval type
164                random_var = self.interval_vars[random_interval_keys[i]]
165
166                if random_var == 0:
167                    # if there is no value to add/subtract randomly, continue on
168                    final_intervals[i] = base_intervals
169
170                else:
171                    # otherwise make num_trials amount of random integers and add them to the final_intervals
172                    random_interval_arr = np.random.randint(
173                        -random_var, random_var, num_trials
174                    )
175
176                    final_intervals[i] = np.add(base_intervals, random_interval_arr)
177
178            # assign results for respective types to the model for storage and later use
179            self.ITI_intervals_final = final_intervals[0]
180            self.TTC_intervals_final = final_intervals[1]
181            self.sample_intervals_final = final_intervals[2]
182
183            logger.info("Random intervals created.")
184        except Exception as e:

186            raise

197    def create_trial_blocks(self) -> List:
198        """
199        This function generates all possible pairings that can occur in ONE TRIAL BLOCK given the input
200        valve stimuli. For example, if we have valves 1 and 5 containing substance 1, and valves 2 and 6
201        containing substance 2, this method will generate a list of two pairings that can occur in one trial block.
202        We need each pairing to complete a block, so we generate [(substance 1, substance 2), (substance 2, substance 1)]
203        as our output. This covers each possible pairing.          (valve 1),   (valve 6)      (valve 2),    (valve 5)
204                                                0 indexing ---->     (0)          (5)             (1)         (4)
205        """
206        try:
207            # creating pairs list which stores all possible pairs
208            pairs = []
209            num_stimuli = self.exp_var_entries["Num Stimuli"]
210            block_sz = num_stimuli // 2
211
212            stimuli_names = list(self.stimuli_data.stimuli_vars.values())
213
214            for i in range(block_sz):
215                pair_index = self.get_paired_index(i, num_stimuli)
216                pairs.append((stimuli_names[i], stimuli_names[pair_index]))
217
218            return pairs
219        except Exception as e:
220            logger.error(f"Error creating trial blocks: {e}")
221            raise

223    def generate_pairs(self, pairs) -> Tuple[list, list]:
224        """
225        This method takes the possible pairings given by create trial blocks and generates
226        which substance will be on a given port for each trial of the experiment. To
227        ensure pseudo randomness, we utilize np.random.shuffle on each pairing.
228        """
229        try:
230            pseudo_random_lineup: List[tuple] = []
231
232            stimulus_1: List[str] = []
233            stimulus_2: List[str] = []
234
235            for _ in range(self.exp_var_entries["Num Trial Blocks"]):
236                pairs_copy = [tuple(pair) for pair in pairs]
237                np.random.shuffle(pairs_copy)
238                pseudo_random_lineup.extend(pairs_copy)
239
240            for pair in pseudo_random_lineup:
241                stimulus_1.append(pair[0])
242                stimulus_2.append(pair[1])
243
244            return stimulus_1, stimulus_2
245        except Exception as e:
246            logger.error(f"Error generating pairs: {e}")
247            raise

249    def build_frame(
250        self,
251        stimuli_1,
252        stimuli_2,
253    ) -> None:
254        num_stimuli = self.exp_var_entries["Num Stimuli"]
255        num_trials = self.exp_var_entries["Num Trials"]
256        num_trial_blocks = self.exp_var_entries["Num Trial Blocks"]
257
258        # each block must contain each PAIR -> num pairs = num_stim / 2
259        block_size = int(num_stimuli / 2)
260        try:
261            data = {
262                "Trial Block": np.repeat(
263                    range(1, num_trial_blocks + 1),
264                    block_size,
265                ),
266                "Trial Number": np.repeat(range(1, num_trials + 1), 1),
267                "Port 1": stimuli_1,
268                "Port 2": stimuli_2,
269                "Port 1 Licks": np.full(num_trials, np.nan),
270                "Port 2 Licks": np.full(num_trials, np.nan),
271                "ITI": self.ITI_intervals_final,
272                "TTC": self.TTC_intervals_final,
273                "SAMPLE": self.sample_intervals_final,
274                "TTC Actual": np.full(num_trials, np.nan),
275            }
276
277            self.program_schedule_df = pd.DataFrame(data)
278
279            logger.info("Initialized stimuli dataframe.")
280        except Exception as e:
281            logger.debug(f"Error Building Stimuli Frame: {e}.")
282            raise

293    @staticmethod
294    def get_paired_index(i, num_stimuli):
295        match num_stimuli:
296            case 2:
297                # if num stim is 2, we are only running the same stimulus on both sides,
298                # we will only see i=0 here, and its paired with its sibling i=4
299                return i + 4
300            case 4:
301                # if num stim is 4, we are only running 2 stimuli per side,
302                # we will see i=0 and i =1 here, paired with siblings i=5 & i=4 respectively
303                match i:
304                    case 0:  # valve 1
305                        return i + 5  # (5) / valve 6
306                    case 1:  # valve 2
307                        return i + 3  # (4) / valve 5
308            case 8:
309                match i:
310                    case 0:
311                        return i + 5
312                    case 1:
313                        return i + 3
314                    case 2:
315                        return i + 5
316                    case 3:
317                        return i + 3
318            case _:
319                return None
