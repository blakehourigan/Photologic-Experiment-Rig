141        # fill the tkinter entry boxes with their default values as configured in self.exp_data
142        for key in self.tkinter_entries.keys():
143            self.tkinter_entries[key].set(self.exp_data.get_default_value(key))
144
145        # we add traces to all of our tkinter variables so that on value update
146        # we sent those updates to the corresponding model (data storage location)
147        for key, value in self.tkinter_entries.items():
148            value.trace_add(
149                "write",
150                lambda *args, key=key, value=value: self.exp_data.update_model(
151                    key, GUIUtils.safe_tkinter_get(value)
152                ),
153            )
154
155        # tkinter variables for other variables in the experiment such as num stimuli
156        # or number of trial blocks
157        self.exp_var_entries = {
158            "Num Trial Blocks": tk.IntVar(),

160        }
161
162        # fill the exp_var entry boxes with their default values as configured in self.exp_data
163        for key in self.exp_var_entries.keys():
164            self.exp_var_entries[key].set(self.exp_data.get_default_value(key))
